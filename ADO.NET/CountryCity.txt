CREATE TABLE Countries
(
	id int not null identity(1,1),
	name varchar(32),
	CONSTRAINT CountConstr primary key (id)
)
CREATE Table Cities
(
	id int not null identity(1,1),
	name varchar(32),
	id_country int,
	CONSTRAINT ConstrCity foreign key(id_country) REFERENCES Countries(id)
)
